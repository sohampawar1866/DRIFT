    ```python
    """E2E-01: full chain on a real MARIDA patch + real CMEMS+ERA5 slice in < 15 s.

    Measurement discipline (RESEARCH Pitfall 9):
      - Imports happen at module load, BEFORE any timing starts.
      - Matplotlib + fonts are warmed by importing backend.mission.export at
        module load (that module already runs a 1x1 figure warm-up).
      - A "pre-flight" call of export_pdf on a tiny fixture is made inside the
        test fixture, also before t0 — this warms reportlab and the coastline
        cache.
      - t0 = time.perf_counter() AFTER warm-up, immediately before run_inference.
      - t_total is the sum of the four stage durations; the < 15 s assertion
        is against t_total, not pytest wall-clock.

    Fallback logic when our_real checkpoint is missing:
      - Test skips-with-warning if neither our_real checkpoint nor a MARIDA
        patch is on disk.
      - If only the checkpoint is missing, the test runs against 'dummy'
        weights and XFAILs — the chain is exercised end-to-end but the
        < 15 s number is not considered authoritative.
    """
    from __future__ import annotations

    import glob
    import logging
    import time
    from pathlib import Path

    import pytest

    # Import everything BEFORE any timing — these imports are the thing we
    # explicitly do NOT want inside the perf_counter window.
    from backend.core.config import Settings
    from backend.core.schemas import (
        DetectionFeatureCollection,
        ForecastEnvelope,
        MissionPlan,
    )
    from backend.ml.inference import run_inference
    from backend.ml.weights import OUR_REAL_CANDIDATES
    from backend.mission.export import export_gpx, export_geojson, export_pdf
    from backend.mission.planner import plan_mission
    from backend.physics.tracker import forecast_drift
    from tests.fixtures.synthetic_mission import make_mission_plan

    log = logging.getLogger("drift.e2e_test")

    CMEMS_PATH = Path("data/env/cmems_currents_72h.nc")
    ERA5_PATH = Path("data/env/era5_winds_72h.nc")
    BUDGET_TOTAL_S = 15.0
    BUDGET_STAGE_S = {"inference": 6.0, "forecast": 5.0, "mission": 1.0, "export": 3.0}


    def _find_marida_patch() -> Path | None:
        candidates = sorted(glob.glob("MARIDA/patches/**/S2_*_0.tif", recursive=True))
        return Path(candidates[0]) if candidates else None


    def _warm_pdf(tmp_path: Path) -> None:
        """Pre-flight PDF call — warms matplotlib fonts + reportlab + coastline cache."""
        export_pdf(make_mission_plan(n_waypoints=3), None, tmp_path / "warm.pdf")


    @pytest.fixture(scope="module")
    def cfg_real() -> Settings:
        base = Settings()
        if any(p.exists() for p in OUR_REAL_CANDIDATES):
            return base.model_copy(update={
                "ml": base.ml.model_copy(update={"weights_source": "our_real"})
            })
        log.warning("our_real checkpoint not present — falling back to 'dummy' for E2E test")
        return base


    def test_e2e_chain_under_15s(cfg_real: Settings, tmp_path: Path) -> None:
        patch = _find_marida_patch()
        if patch is None:
            pytest.skip("no MARIDA patch available on disk")
        if not CMEMS_PATH.exists() or not ERA5_PATH.exists():
            pytest.skip(f"env NetCDFs missing: {CMEMS_PATH}, {ERA5_PATH}")

        # Warm-up BEFORE timing window (RESEARCH Pitfall 9).
        _warm_pdf(tmp_path)

        timings: dict[str, float] = {}

        # Stage 1: inference
        t = time.perf_counter()
        detections = run_inference(patch, cfg_real)
        timings["inference"] = time.perf_counter() - t
        DetectionFeatureCollection.model_validate(detections.model_dump())

        # Stage 2: forecast
        t = time.perf_counter()
        forecast = forecast_drift(detections, cfg_real)
        timings["forecast"] = time.perf_counter() - t
        ForecastEnvelope.model_validate(forecast.model_dump())

        # Stage 3: mission
        t = time.perf_counter()
        mission = plan_mission(forecast, 200.0, 8.0, (78.9, 9.2), cfg_real)
        timings["mission"] = time.perf_counter() - t
        MissionPlan.model_validate(mission.model_dump())

        # Stage 4: export (GPX + GeoJSON + PDF all together — this is the demo path)
        t = time.perf_counter()
        export_gpx(mission, tmp_path / "out.gpx")
        export_geojson(mission, tmp_path / "out.geojson")
        export_pdf(mission, forecast, tmp_path / "out.pdf")
        timings["export"] = time.perf_counter() - t

        total = sum(timings.values())
        report = (
            f"\nE2E timing (s):\n"
            f"  inference: {timings['inference']:.2f} (budget {BUDGET_STAGE_S['inference']})\n"
            f"  forecast:  {timings['forecast']:.2f} (budget {BUDGET_STAGE_S['forecast']})\n"
            f"  mission:   {timings['mission']:.2f} (budget {BUDGET_STAGE_S['mission']})\n"
            f"  export:    {timings['export']:.2f} (budget {BUDGET_STAGE_S['export']})\n"
            f"  TOTAL:     {total:.2f} (budget {BUDGET_TOTAL_S})"
        )
        print(report)

        # Per-stage soft checks (log but don't fail individually — total is the hard gate)
        for stage, elapsed in timings.items():
            if elapsed > BUDGET_STAGE_S[stage]:
                log.warning(f"stage {stage} over per-stage budget: {elapsed:.2f}s > {BUDGET_STAGE_S[stage]}s")

        # Hard gate: total < 15 s (PRD Core Value)
        # If running on fallback 'dummy' weights because user handoff missed, we
        # xfail rather than fail — the number isn't authoritative then.
        using_real = cfg_real.ml.weights_source == "our_real"
        if not using_real:
            pytest.xfail(
                f"E2E ran on dummy weights (our_real checkpoint missing). "
                f"Total {total:.2f}s — budget cannot be authoritatively measured."
            )
        assert total < BUDGET_TOTAL_S, report


    def test_e2e_stage_outputs_schema_valid(cfg_real: Settings, tmp_path: Path) -> None:
        """Regression: every stage's output model_dump_json round-trips through model_validate_json."""
        patch = _find_marida_patch()
        if patch is None:
            pytest.skip("no MARIDA patch available on disk")
        if not CMEMS_PATH.exists() or not ERA5_PATH.exists():
            pytest.skip(f"env NetCDFs missing")

        detections = run_inference(patch, cfg_real)
        rd = DetectionFeatureCollection.model_validate_json(detections.model_dump_json())
        assert rd == detections

        forecast = forecast_drift(detections, cfg_real)
        rf = ForecastEnvelope.model_validate_json(forecast.model_dump_json())
        assert rf == forecast

        mission = plan_mission(forecast, 200.0, 8.0, (78.9, 9.2), cfg_real)
        rm = MissionPlan.model_validate_json(mission.model_dump_json())
        assert rm == mission
    ```
