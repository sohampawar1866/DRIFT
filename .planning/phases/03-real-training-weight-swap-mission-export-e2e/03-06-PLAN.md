    """Pre-bake the 4 demo AOIs at H+28 per D-16.

    Runs full chain (our_real weights, CPU-only, fixed seeds) for each AOI and
    writes {aoi}_{detections|forecast|mission}.json + MANIFEST.json to
    data/prebaked/. MANIFEST records parity hashes that
    tests/test_prebake_parity.py later compares against a fresh live run.

    Usage:
        python scripts/prebake_demo.py [--allow-dirty] [--aoi AOI ...]
    """
    from __future__ import annotations

    import argparse
    import json
    import os
    import random
    import subprocess
    import sys
    from datetime import datetime, timezone
    from pathlib import Path

    import numpy as np
    import torch

    from backend.core.config import Settings
    from scripts.parity_hash import parity_hash
    from scripts.run_full_chain_real import run_chain

    PREBAKE_DIR = Path("data/prebaked")
    SEED = 1410

    AOIS: list[tuple[str, str, tuple[float, float]]] = [
        ("gulf_of_mannar",        "MARIDA/patches/**/S2_*_0.tif", (78.9, 9.2)),
        ("mumbai_offshore",       "MARIDA/patches/**/S2_*_0.tif", (72.8, 18.9)),
        ("bay_of_bengal_mouth",   "MARIDA/patches/**/S2_*_0.tif", (87.5, 21.5)),
        ("arabian_sea_gyre_edge", "MARIDA/patches/**/S2_*_0.tif", (65.0, 15.0)),
    ]


    def _seed_everything(seed: int = SEED) -> None:
        os.environ["PYTHONHASHSEED"] = "0"
        random.seed(seed)
        np.random.seed(seed)
        torch.manual_seed(seed)
        torch.set_num_threads(1)
        try:
            torch.use_deterministic_algorithms(True, warn_only=True)
        except Exception as e:
            print(f"WARN: use_deterministic_algorithms not fully available: {e}",
                  file=sys.stderr)


    def _git_sha() -> str:
        try:
            return subprocess.check_output(["git", "rev-parse", "HEAD"], text=True).strip()
        except Exception:
            return "unknown"


    def _git_dirty() -> bool:
        try:
            return bool(subprocess.check_output(
                ["git", "status", "--porcelain"], text=True).strip())
        except Exception:
            return False


    def _resolve_tile(aoi: str, glob_pattern: str) -> Path:
        import glob as _g
        cand = sorted(_g.glob(glob_pattern, recursive=True))
        if not cand:
            raise FileNotFoundError(
                f"No MARIDA tile for {aoi} under {glob_pattern}. "
                "Edit AOIS in scripts/prebake_demo.py to point at a real tile.")
        return Path(cand[0])


    def main() -> int:
        ap = argparse.ArgumentParser()
        ap.add_argument("--allow-dirty", action="store_true",
                        help="Skip git-clean check (RESEARCH Pitfall 10)")
        ap.add_argument("--aoi", action="append", default=None,
                        help="Restrict to named AOIs (repeatable).")
        args = ap.parse_args()

        if _git_dirty() and not args.allow_dirty:
            print("ERROR: git working tree dirty. Commit/stash or pass --allow-dirty.",
                  file=sys.stderr)
            return 2

        PREBAKE_DIR.mkdir(parents=True, exist_ok=True)
        _seed_everything()

        cfg = Settings()
        cfg = cfg.model_copy(update={
            "ml": cfg.ml.model_copy(update={"weights_source": "our_real"})})

        work = AOIS if not args.aoi else [a for a in AOIS if a[0] in set(args.aoi)]
        entries: list[dict] = []
        for aoi, tile_glob, origin in work:
            _seed_everything()  # re-seed before each AOI for independence
            tile = _resolve_tile(aoi, tile_glob)
            print(f"[{aoi}] tile={tile} origin={origin}")
            result = run_chain(aoi, tile, origin, cfg, no_fallback=True)
            for stage_name, model in (
                ("detections", result["detections"]),
                ("forecast",   result["forecast"]),
                ("mission",    result["mission"]),
            ):
                out = PREBAKE_DIR / f"{aoi}_{stage_name}.json"
                out.write_text(model.model_dump_json(indent=2), encoding="utf-8")
                h = parity_hash(model)
                entries.append({"aoi": aoi, "stage": stage_name,
                                "path": out.name, "parity_hash": h})
                print(f"  {stage_name}: {out.name} hash={h[:16]}...")

        manifest = {
            "generated_at": datetime.now(timezone.utc).isoformat(),
            "git_sha": _git_sha(),
            "weights_source": cfg.ml.weights_source,
            "determinism": {"torch_seed": SEED, "numpy_seed": SEED,
                            "random_seed": SEED, "python_hash_seed": 0,
                            "num_threads": 1, "deterministic_algorithms": True},
            "entries": entries,
        }
        (PREBAKE_DIR / "MANIFEST.json").write_text(
            json.dumps(manifest, indent=2), encoding="utf-8")
        print(f"wrote {PREBAKE_DIR/'MANIFEST.json'} "
              f"({len(entries)} entries, git_sha={manifest['git_sha'][:10]})")
        return 0


    if __name__ == "__main__":
        raise SystemExit(main())
    ```

    """E2E-02 close: byte-identical parity between prebake and a fresh live run.

    Runs the same chain on the same AOIs with the same seeds; each stage's
    parity_hash must equal the one recorded in MANIFEST.json.
    """
    from __future__ import annotations

    import json
    import os
    import random
    from pathlib import Path

    import numpy as np
    import pytest
    import torch

    from backend.core.config import Settings
    from scripts.parity_hash import parity_hash
    from scripts.run_full_chain_real import run_chain

    PREBAKE_DIR = Path("data/prebaked")
    MANIFEST_PATH = PREBAKE_DIR / "MANIFEST.json"
    SEED = 1410

    AOI_ORIGINS = {
        "gulf_of_mannar": (78.9, 9.2),
        "mumbai_offshore": (72.8, 18.9),
        "bay_of_bengal_mouth": (87.5, 21.5),
        "arabian_sea_gyre_edge": (65.0, 15.0),
    }


    def _seed_everything() -> None:
        os.environ["PYTHONHASHSEED"] = "0"
        random.seed(SEED)
        np.random.seed(SEED)
        torch.manual_seed(SEED)
        torch.set_num_threads(1)
        try:
            torch.use_deterministic_algorithms(True, warn_only=True)
        except Exception:
            pass


    def _manifest():
        if not MANIFEST_PATH.exists():
            pytest.skip("MANIFEST.json missing — run scripts/prebake_demo.py first")
        return json.loads(MANIFEST_PATH.read_text(encoding="utf-8"))


    @pytest.mark.integration
    def test_prebake_parity_live_equals_manifest():
        manifest = _manifest()
        if manifest.get("weights_source") != "our_real":
            pytest.skip("MANIFEST was not produced with our_real weights")

        import glob
        tiles = sorted(glob.glob("MARIDA/patches/**/S2_*_0.tif", recursive=True))
        if not tiles:
            pytest.skip("no MARIDA tile available for live parity run")
        tile = Path(tiles[0])

        cfg = Settings()
        cfg = cfg.model_copy(update={
            "ml": cfg.ml.model_copy(update={"weights_source": "our_real"})})

        # Index: {(aoi,stage): hash}
        expected = {(e["aoi"], e["stage"]): e["parity_hash"] for e in manifest["entries"]}
        aois_in_manifest = sorted({e["aoi"] for e in manifest["entries"]})

        mismatches: list[tuple[str, str, str, str]] = []
        for aoi in aois_in_manifest:
            if aoi not in AOI_ORIGINS:
                continue
            _seed_everything()
            result = run_chain(aoi, tile, AOI_ORIGINS[aoi], cfg, no_fallback=True)
            for stage, model in (
                ("detections", result["detections"]),
                ("forecast",   result["forecast"]),
                ("mission",    result["mission"]),
            ):
                key = (aoi, stage)
                if key not in expected:
                    continue
                got = parity_hash(model)
                if got != expected[key]:
                    mismatches.append((aoi, stage, expected[key], got))

        assert not mismatches, (
            "Parity broken on:\n" +
            "\n".join(f"  {a}/{s}: manifest={m[:16]} live={g[:16]}"
                     for a, s, m, g in mismatches)
        )


    def test_manifest_shape_sane():
        m = _manifest()
        for key in ("generated_at", "git_sha", "weights_source",
                    "determinism", "entries"):
            assert key in m, f"MANIFEST missing {key}"
        for e in m["entries"]:
            assert set(e.keys()) >= {"aoi", "stage", "path", "parity_hash"}
            assert len(e["parity_hash"]) == 64
            assert (PREBAKE_DIR / e["path"]).exists(), e["path"]
    ```
